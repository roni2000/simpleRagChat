# Postgres DB User

### Create new user


```
CREATE ROLE "pocUser1" LOGIN;
SELECT rolname FROM pg_roles;
ALTER ROLE "pocUser1" WITH PASSWORD 'ThisIsSimplePassword123';

GRANT CONNECT ON DATABASE poc1 TO "pocUser1";
GRANT USAGE ON SCHEMA public TO "pocUser1";
GRANT CREATE ON SCHEMA public TO "pocUser1";
GRANT SELECT, INSERT, UPDATE, DELETE ON ALL TABLES IN SCHEMA public TO "pocUser1";
GRANT USAGE, SELECT, UPDATE ON ALL SEQUENCES IN SCHEMA public TO "pocUser1";
ALTER DEFAULT PRIVILEGES IN SCHEMA public GRANT SELECT, INSERT, UPDATE, DELETE ON TABLES TO "pocUser1";
ALTER DEFAULT PRIVILEGES IN SCHEMA public GRANT USAGE, SELECT, UPDATE ON SEQUENCES TO "pocUser1";
```
